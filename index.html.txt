        document.addEventListener('DOMContentLoaded', function() {
            const btnNewUser = document.getElementById('btnNewUser');
            const btnExistingUser = document.getElementById('btnExistingUser');
            const mainContent = document.getElementById('mainContent');
            const newUserContent = document.getElementById('newUserContent');
            const existingUserContent = document.getElementById('existingUserContent');
            const userTypeBtns = document.querySelectorAll('.user-type-btn');

            function setActiveButton(activeBtn) {
                userTypeBtns.forEach(btn => {
                    btn.classList.remove('bg-teal-600', 'text-white', 'border-teal-600');
                    btn.classList.add('bg-white', 'text-slate-800', 'border-slate-300', 'hover:bg-teal-50', 'hover:border-teal-400');
                });
                activeBtn.classList.add('bg-teal-600', 'text-white', 'border-teal-600');
                activeBtn.classList.remove('bg-white', 'text-slate-800', 'border-slate-300', 'hover:bg-teal-50', 'hover:border-teal-400');
            }

            btnNewUser.addEventListener('click', () => {
                mainContent.classList.remove('content-hidden');
                newUserContent.classList.remove('content-hidden');
                existingUserContent.classList.add('content-hidden');
                setActiveButton(btnNewUser);
            });

            btnExistingUser.addEventListener('click', () => {
                mainContent.classList.remove('content-hidden');
                existingUserContent.classList.remove('content-hidden');
                newUserContent.classList.add('content-hidden');
                setActiveButton(btnExistingUser);
            });

            // Accordion functionality
            const accordionItems = document.querySelectorAll('.accordion-item');
            accordionItems.forEach(item => {
                const header = item.querySelector('.accordion-header');
                const content = item.querySelector('.accordion-content');
                const icon = item.querySelector('.accordion-icon');
                header.addEventListener('click', () => {
                    const isExpanded = content.style.maxHeight && content.style.maxHeight !== '0px';
                    
                    accordionItems.forEach(i => {
                        i.querySelector('.accordion-content').style.maxHeight = '0px';
                        i.querySelector('.accordion-icon').textContent = '＋';
                        i.querySelector('.accordion-icon').classList.remove('rotate-45');
                    });

                    if (!isExpanded) {
                        content.style.maxHeight = content.scrollHeight + 'px';
                        icon.textContent = '－';
                        icon.classList.add('rotate-45');
                    }
                });
            });

            // Short-term support tabs
            const shortTermTabs = document.querySelectorAll('.short-term-tab');
            const shortTermContents = document.querySelectorAll('.short-term-content');
            shortTermTabs.forEach(tab => {
                tab.addEventListener('click', () => {
                    shortTermTabs.forEach(t => t.classList.replace('tab-active', 'tab-inactive'));
                    tab.classList.replace('tab-inactive', 'tab-active');

                    const targetId = tab.getAttribute('data-target');
                    shortTermContents.forEach(content => {
                        if (content.id === targetId) {
                            content.classList.remove('content-hidden');
                        } else {
                            content.classList.add('content-hidden');
                        }
                    });
                });
            });

            // Fee Calculator
            const pensionStatus = document.getElementById('pensionStatus');
            const serviceType = document.getElementById('serviceType');
            const feePercentage = document.getElementById('feePercentage');
            const feeDescription = document.getElementById('feeDescription');

            const feeMatrix = {
                full: { clinical: 0, living: 5, home: 17.5 },
                part: { clinical: 0, living: '5% - 50%', home: '17.5% - 80%' },
                self: { clinical: 0, living: 50, home: 80 }
            };
            
            const descriptionMatrix = {
                 clinical: "臨床支援服務由政府全額資助。",
                 living: "此類服務有助於維持獨立生活。",
                 home: "此類服務與日常居家生活相關。"
            };

            function updateFee() {
                const status = pensionStatus.value;
                const type = serviceType.value;
                const fee = feeMatrix[status][type];
                const desc = descriptionMatrix[type];

                if (typeof fee === 'string') {
                    feePercentage.textContent = fee;
                } else {
                    feePercentage.textContent = `${fee}%`;
                }
                feeDescription.textContent = desc;
            }

            pensionStatus.addEventListener('change', updateFee);
            serviceType.addEventListener('change', updateFee);
            updateFee();

            // Funding Levels Chart
            const ctx = document.getElementById('fundingLevelsChart').getContext('2d');
            const fundingData = {
                labels: ['等級 1', '等級 2', '等級 3', '等級 4', '等級 5', '等級 6', '等級 7', '等級 8'],
                datasets: [{
                    label: '季度預算 (澳元)',
                    data: [2674.18, 3995.42, 5479.94, 7386.33, 9883.76, 11989.35, 14530.53, 19427.25],
                    backgroundColor: '#ccfbf1',
                    borderColor: '#0d9488',
                    borderWidth: 2,
